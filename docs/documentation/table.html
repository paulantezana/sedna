let onConfirmDelete = ()=>{
    SnModal.confirm({
        title: 'Are you sure delete this task?',
        content: 'Some descriptions',
        okText: 'Yes',
        okType: 'error',
        cancelText: 'No',
        onOk() {
            console.log('onOk')
        },
        onCancel() {
            console.log('onCancel')
        }
    });
}
